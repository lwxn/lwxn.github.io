    第一种暴力解：

    class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int>ans;
        for(int i = 0;i<nums.size();i++)
        {
            for(int j = i+1;j<nums.size();j++)
            {
                if(nums[i] + nums[j] == target)
                {
                    ans.push_back(i);
                    ans.push_back(j);
                    return ans;
                }
            }
        }
        return ans;
    }
};

    第二种，使用hashmap

    class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int>Map;
        vector<int>ans;

        for(int i = 0;i<nums.size();i++)
        {
            auto it = Map.find(target-nums[i]);
            if(it != Map.end())
            {
                return {i,it->second};
            }
            Map[nums[i]] = i;
        }
        return ans;
    }
};
