    7 Reverse Integer

    January 18, 2021 
    class Solution {
public:
    int reverse(int x) {
      long X = x;
      int flag = 0;
      if(X < 0)
      {
          X = -X;
          flag = 1;
      }

      long ans = 0;
      string s = to_string(X);
      for(int i = s.size()-1;i>=0;i--) ans = ans*10 + s[i] - '0';

      ans = flag ? -ans : ans;
      int Max = 2147483647;
      int Min = -2147483648;
      if(ans <= Max && ans >= Min) return ans;
      else return 0; 
    }
};
