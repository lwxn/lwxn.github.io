    12 Integer to Roman

    January 18, 2021 
    class Solution {
public:
    string intToRoman(int num) {
        int nums[13] = {1,4,5,9,10,40,50,90,100,400,500,900,1000};
        string alphas[13] = {"I","IV","V","IX","X","XL","L","XC","C","CD","D","CM","M"},ans = "";

        for(int i = 12;i>=0 && num;i--)
        {
            while(num >= nums[i])
            {
                ans = ans + alphas[i];
                num -= nums[i];
            }
        }
        
        return ans;

    }
};```
