    6 ZigZag Conversion

    January 18, 2021 
    class Solution {
public:
    string convert(string s, int numRows) {
        if(s.size() == 0) return "";
        int n = s.size();
        vector<vector<char>>Map(numRows,vector<char>(n,' '));

        int mode = 0;
        int i = 0;
        int x = -1,y = 0;
        while(i<n)
        {
            //竖排
            if(mode == 0)
            {
                
                for(int j = 0;j<numRows && i<n;j++)
                {
                    if(j == 0) x++;
                    Map[j][x] = s[i];
                    y = j;
                    i++;
                }
            }
            else
            {
                for(int j = 1;j<numRows-1 && i<n;j++)
                {
                    x++;
                    y--;
                    Map[y][x] =  s[i];
                    i++;
                }
            }
            mode = (mode+1)%2;
        }

        string ans = "";
        for(int i = 0;i<numRows;i++)
        {
            for(int j = 0;j<n;j++)
            {
                if(Map[i][j] != ' ')
                    ans.push_back(Map[i][j]);
            }
        }
        return ans;
    }
};
