    3 Longest Substring Without Repeating Characters

    January 17, 2021

    滑动窗口

    class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int ans = 0,i = 0,j = 1;
        if(s.size() == 1) ans = 1;
        unordered_map<char,int>Map;
        Map[s[i]] = i;
        
        while(i<s.size() && j<s.size())
        {
            auto it = Map.find(s[j]);
            if(it != Map.end() && it->second >= i)
            {
                ans = max(ans,j-i);
                i = Map[s[j]] + 1;
            }
            else 
            {
                Map[s[j]] = j;
                ans = max(ans,j-i+1);
                j++;
            }
        }
        return ans;
    }
};
